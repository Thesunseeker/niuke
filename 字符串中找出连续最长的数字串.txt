#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main()
{
	string s1, s2, s3;
	cin >> s1;
	for (unsigned long i = 0; i < s1.length(); i++)
	{
		if (s1[i] >= '0' && s1[i] <= '9')
			s3 += s1[i];
		else{
			if (s2.length() < s3.length())	s2 = s3;
			s3 = "";
		}
	}
	cout << s2;
	return 0;
}
